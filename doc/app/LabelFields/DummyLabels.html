def actual_count
  values.to_h.reduce(0) do |result, (k,v)|
    result += (PLACEHOLDERS.include?(k) ? 1 : 2)
  end

    to_h() click to toggle source

    produces a hash from the open struct. for any label that is not a header or footer will produce two copies which will be appended to the hash as an array of key body e.g.

    dummy_labels = DummyLabels.new({label_1: [:field_1, :field_2], label_2: [:field_3, :field_4], header: [:field_5, :field_6], footer: [:field_7, :field_8]})
dummy_labels.to_h =>
{ header: {field_5: "field_5", field_6: "field_6"}, footer: {field_7: "field_7", field_8: "field_8"}, body: [{ label_1: {field_1: "field_1", field_2: "field_2"}, label_2: {field_3: "field_3", field_4: "field_4"} }, { label_1: {field_1: "field_1", field_2: "field_2"}, label_2: {field_3: "field_3", field_4: "field_4"} }]}
    # File app/models/label_fields.rb, line 88
def to_h
  values.to_h.slice(*PLACEHOLDERS).merge({body: labels}).with_indifferent_access
end
